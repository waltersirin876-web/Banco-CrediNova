<script>
    let saldoReal = 24500.00;
    let vaultHidden = false;

    // Lógica Bóveda
    function toggleVault() {
        vaultHidden = !vaultHidden;
        const display = document.getElementById('display-saldo');
        if (vaultHidden) {
            display.innerText = "Q ••••••••";
            display.style.letterSpacing = "3px";
        } else {
            display.innerText = `Q ${saldoReal.toLocaleString('en-US', {minimumFractionDigits: 2})}`;
            display.style.letterSpacing = "0px";
        }
    }

    // Lógica QR
    function generarQR() {
        const monto = document.getElementById('monto-qr').value;
        const area = document.getElementById('qr-area');
        const container = document.getElementById('qr-container');
        if (monto > 0) {
            area.style.display = 'block';
            container.innerHTML = "";
            new QRCode(container, { text: `PAGO|${monto}`, width: 120, height: 120, colorDark: "#1b4332" });
        } else { area.style.display = 'none'; }
    }

    // Lógica Chatbot
    function toggleChat() {
        const win = document.getElementById('chat-window');
        win.style.display = win.style.display === 'flex' ? 'none' : 'flex';
    }

    function sendChat() {
        const input = document.getElementById('user-msg');
        const content = document.getElementById('chat-content');
        if (!input.value) return;

        content.innerHTML += `<div style="text-align:right; margin-bottom:10px;"><b>Tú:</b> ${input.value}</div>`;
        
        // Respuestas automáticas simples
        let respuesta = "Lo siento, no entiendo esa consulta. Prueba con 'horario', 'préstamo' o 'sucursal'.";
        const msg = input.value.toLowerCase();
        
        if (msg.includes("horario")) respuesta = "Atendemos de lunes a viernes de 9:00 AM a 6:00 PM.";
        if (msg.includes("préstamo")) respuesta = "Contamos con tasas desde el 12% anual. ¡Usa nuestro simulador arriba!";
        if (msg.includes("sucursal")) respuesta = "Nuestra sede central está en la Zona Viva, Edificio El Bosque.";

        setTimeout(() => {
            content.innerHTML += `<div class="bot-msg"><b>Bot:</b> ${respuesta}</div>`;
            content.scrollTop = content.scrollHeight;
        }, 600);
        
        input.value = "";
    }
</script>
